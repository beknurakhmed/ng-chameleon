Get up and running in under a minute.

terminal
$ npm install ng-chameleon
# Add components to your project
$ ng g ng-chameleon:add button card input
app.component.html 
<ch-button variant="solid" colorScheme="primary">
  Click me
</ch-button>

<!-- Style props (Chakra-like) -->
<div chBox bg="primary.500" p="4" rounded="lg">
  <span chText fontSize="xl">Styled</span>
</div>
app.component.ts 
import {{ '{' }} ChameleonThemeService {{ '}' }} from 'ng-chameleon';

// Switch theme at runtime — zero rebuild!
this.themeService.setTheme('material');
this.themeService.setTheme('shadcn');
this.themeService.setTheme('chakra');
this.themeService.setTheme('antd');
